CREATE DATABASE ferryDB;

CREATE TABLE ferryTABLE(
`Trip ID` VARCHAR(10) PRIMARY KEY,
`Body No.` VARCHAR(10),
`Route` VARCHAR(10),
`Location` VARCHAR(15),
`ETA` VARCHAR(15),
`Seats Available` VARCHAR(10),
`Status` VARCHAR(10),
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE passengerTABLE(
`Trip ID` VARCHAR(10),
`NAME` VARCHAR(50),
`AGE` VARCHAR(3),
`GENDER` VARCHAR(10),
`ADDRESS` VARCHAR(200),
`CONTACT #` VARCHAR(15),
`DESTINATION` VARCHAR(20),
`Ticket Bought` TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE ferryDeletedTrip(
`Trip ID` VARCHAR(10),
`Body No.` VARCHAR(10),
`Route` VARCHAR(10),
`Location` VARCHAR(15),
`ETA` VARCHAR(15),
`Seats Available` VARCHAR(10),
`Status` VARCHAR(10),
deleted_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE deletedPassengerData(
`Trip ID` VARCHAR(10),
`NAME` VARCHAR(50),
`AGE` VARCHAR(3),
`GENDER` VARCHAR(10),
`ADDRESS` VARCHAR(200),
`CONTACT #` VARCHAR(15),
`DESTINATION` VARCHAR(20),
`DELETED AT` TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
